## 반복문
  ~~~php
  $record = array(1, "사람1", 26, "Good", "학생");

  // for문
  for($i = 0 ; $i < count($record); $i++){
    echo $record[$i]."<br/>";
  }

  // while 문
  $cnt = 0;
  while($cnt < count($record)){
    echo $record[$cnt]."<br>";
    $cnt++;
  }
  ~~~
